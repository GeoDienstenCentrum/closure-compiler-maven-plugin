6   import java.util.List;

8   import org.apache.maven.plugin.AbstractMojo;
9   import org.apache.maven.plugin.MojoExecutionException;
10  import org.apache.maven.plugin.MojoFailureException;
11  import org.apache.maven.plugin.logging.Log;

13  import com.google.common.base.Charsets;
14  import com.google.common.base.Preconditions;
15  import com.google.common.io.Files;
16  import com.google.javascript.jscomp.CompilationLevel;
17  import com.google.javascript.jscomp.Compiler;
18  import com.google.javascript.jscomp.CompilerOptions;
19  import com.google.javascript.jscomp.JSError;
20  import com.google.javascript.jscomp.Result;
21  import com.google.javascript.jscomp.SourceFile;

23  /**
24   *
25   * @phase generate-sources
26   * @goal compile
27   */
28  public class ClosureCompilerMojo extends AbstractMojo {
29  
30      /**
31       * @parameter @required
32       */
33      private List<Compilation> compilations;
34  
35      @Override
36      public void execute() throws MojoExecutionException, MojoFailureException {
37          Log log = getLog();
38          Preconditions.checkNotNull(this.compilations);
39  
40          for (Compilation compilation : this.compilations) {
41  
42              List<SourceFile> externs
43                      = getSourceFiles(compilation.getExternFiles());
44              List<SourceFile> source
45                      = getSourceFiles(compilation.getSourceFiles());
46  
47              for (String sourceFile : compilation.getExternFiles()) {
48                  log.info("Compiling from extern: " + sourceFile);
49              }
50              for (String sourceFile : compilation.getSourceFiles()) {
51                  log.info("Compiling from source: " + sourceFile);
52              }
53  
54              log.info("Compiling to: " + compilation.getOutputFile());
55              log.info("Compiling with level: "
56                      + compilation.getCompilationLevel());
57  
58              CompilationLevel compilationLevel = null;
59              try {
60                  compilationLevel
61                          = CompilationLevel.valueOf(compilation
62                                  .getCompilationLevel());
63              } catch (IllegalArgumentException e) {
64                  throw new MojoFailureException("Compilation level invalid", e);
65              }
66  
67              CompilerOptions compilerOptions
68                      = compilation.getCompilerOptionsMojo().getCompilerOptions();
69              if (null == compilerOptions) {
70                  log.info("With no compiler options");
71                  compilerOptions = new CompilerOptions();
72              }
73              log.info("Compiler Options:" + compilerOptions);
74  
75              compilationLevel.setOptionsForCompilationLevel(compilerOptions);
76  
77              Compiler compiler = new Compiler();
78              Result result = compiler.compile(externs, source, compilerOptions);
79  
80              // TODO: should log results to a file if desired.
81              for (JSError warning : result.warnings) {
82                  getLog().warn(warning.toString());
83              }
84  
85              for (JSError error : result.errors) {
86                  getLog().error(error.toString());
87              }
88  
89              if (!result.success) {
90                  // TODO: better info?
91                  throw new MojoFailureException("Compilation failure");
92              }
93  
94              try {
95                  Files.createParentDirs(compilation.getOutputFile());
96                  Files.touch(compilation.getOutputFile());
97                  Files.write(compiler.toSource(), compilation.getOutputFile(),
98                          Charsets.UTF_8);
99              } catch (IOException e) {
100                 throw new MojoFailureException(
101                         compilation.getOutputFile() != null ? compilation
102                                 .getOutputFile().toString() : e.getMessage(), e);
103             }
104         }
105     }
106 
107     /**
108      * @return the compilations
109      */
110     public List<Compilation> getCompilations() {
111         return this.compilations;
112     }
113 
114     /**
115      * @param compilations the compilations to set
116      */
117     public void setCompilations(final List<Compilation> compilations) {
118         Preconditions.checkNotNull(compilations,
119                 "The compilations may not be null.");
120         this.compilations = compilations;
121     }
122 
123     /**
124      * Returns a list of SourceFiles if given a list of relative or absolute
125      * file names. Directories are traversed recursively.
126      *
127      * @param filePaths
128      * @return
129      */
130     private List<SourceFile> getSourceFiles(final List<String> filePaths) {
131         Preconditions.checkNotNull(filePaths, "Filepaths are required.");
132 
133         List<SourceFile> externs = new LinkedList<SourceFile>();
134         for (String filepath : filePaths) {
135             File file = new File(filepath);
136             externs.addAll(sourcefileFromFile(file));
137         }
138         return externs;
139     }
140 
141     /**
142      * Fetch a list of SourceFiles from a file. If the file is a directory, it
143      * is traversed recursively. If not, a single SourceFile is returned.
144      *
145      * @param file
146      * @return
147      */
148     private List<SourceFile> sourcefileFromFile(final File file) {
149         Preconditions.checkNotNull(file, "The file may not be null.");
150         Preconditions.checkArgument(file.exists(), "The file must exist.");
151         List<SourceFile> sourceFiles = new LinkedList<SourceFile>();
152         if (file.isDirectory()) {
153             for (File child : file.listFiles()) {
154                 sourceFiles.addAll(sourcefileFromFile(child));
155             }
156         } else {
157             sourceFiles.add(SourceFile.fromFile(file));
158         }
159         return sourceFiles;
160     }
